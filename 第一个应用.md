以下是一个简单的合约，您可以在此合约中获取、增加和减少计数。

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.17;

contract Counter {
    uint public count; // 公开变量，用于存储计数

    // 函数，用于获取当前计数
    function get() public view returns (uint) {
        return count;
    }

    // 函数，将计数加1
    function inc() public {
        count += 1;
    }

    // 函数，将计数减1
    function dec() public {
        // 如果计数已经为0，则此函数将失败
        count -= 1;
    }
}
